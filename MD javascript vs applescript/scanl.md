    // The series of interim values arising
    // from a catamorphism. Parallel to foldl.
    startValue => xs =>
        "GeneratorFunction" !== (
            xs.constructor.constructor.name
        )

                (a, x) => {
                    const v = f(a[0])(x);

                    return [v, a[1].concat(v)];
                }, [startValue, [startValue]]
